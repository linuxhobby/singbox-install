#!/bin/bash

# 工具库：包含颜色定义与格式化输出，用于提升脚本可读性
# 请通过 source ./utils.sh 在主脚本中引入

# 终端颜色定义
export RED='\033[31m'
export GREEN='\033[32m'
export YELLOW='\033[33m'
export BLUE='\033[34m'
export NC='\033[0m' # No Color

# 格式化输出函数
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# 检查依赖项
check_dependencies() {
    local deps=("jq" "sing-box")
    for cmd in "${deps[@]}"; do
        if ! command -v "$cmd" &> /dev/null; then
            log_error "缺少依赖: $cmd，请先安装。"
            exit 1
        fi
    done
    log_info "依赖检查通过。"
}

# 校验 JSON 有效性
validate_json() {
    if ! jq empty "$1" > /dev/null 2>&1; then
        log_error "文件 $1 不是有效的 JSON 格式。"
        return 1
    fi
    return 0
}
